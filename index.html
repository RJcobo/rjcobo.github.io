BakeLang

By Raul Montes

Example: FizzBuzz


Ingredient i        : int;
Ingredient current  : int;

Repeat 1 to 100 {
  Mix i into current;
  If (current % 15 == 0) then {
    Serve "FizzBuzz";
  } Else {
    If (current % 3 == 0) then {
      Serve "Fizz";
    } Else {
      If (current % 5 == 0) then {
        Serve "Buzz";
      } Else {
        Serve current;
      }
    }
  }
}
